# Start production
start = 
        | line ";" start
        | "repeat" I_LITERAL "{" start "}" start
        ;
# Line production
line = "@" modifier line
        | instruction line
        |
        ;

# Instruction production
instruction = "@" operands ;

# Annotation/Property production
modifier =  "annotation" IDENTIFIER
            | "property" IDENTIFIER anyval
            ;

# Operands production
operands =  I_LITERAL operands
            | "," operands
            |
            ;

# Any value production
anyval = I_LITERAL
            | F_LITERAL
            ;
